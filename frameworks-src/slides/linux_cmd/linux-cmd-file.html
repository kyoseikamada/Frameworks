Linuxコマンド
ファイル操作の基本
linux-cmd-01
場所の確認・移動
# 今どこにいるか確認
pwd
# → /home/alice

# ファイル・フォルダの一覧を表示
ls
ls -l        # 詳細（日時・権限）
ls -la       # 隠しファイルも表示（.bashrcなど）

# 移動する
cd /var/log        # 絶対パスで移動
cd Documents       # 相対パスで移動
cd ..              # 1つ上に戻る
cd ~               # ホームに戻る
cd -               # 直前のフォルダに戻る
ファイル・フォルダの作成
# 空ファイルを作る
touch memo.txt

# フォルダを作る
mkdir work
mkdir -p a/b/c   # 深い階層も一度に作れる
コピー・移動・削除
# ファイルをコピー
cp memo.txt memo_backup.txt
cp memo.txt /tmp/            # フォルダへコピー

# フォルダごとコピー（-r が必要）
cp -r work/ work_backup/

# 移動・名前変更（mv は兼用）
mv memo.txt note.txt         # 名前変更
mv note.txt /tmp/            # 場所を移動

# 削除（⚠ ゴミ箱はない・元に戻せない）
rm memo.txt                  # ファイル削除
rm -r old_folder/            # フォルダごと削除
rm -i memo.txt               # 削除前に確認する
ディスク・サイズ確認
# フォルダのサイズを調べる
du -sh Documents/    # -s: 合計 -h: 人が読める単位

# ディスクの空き容量を確認
df -h
⚠ rm -r の注意：確認なしに全削除されます。特に rm -rf / などは絶対に実行しないこと。
