ネットワーク確認コマンド
# IPアドレス確認
ip addr show           # インタフェース一覧
ip addr show eth0      # eth0のみ
hostname -I            # IPアドレスのみ表示

# ルーティング
ip route show
ip route get 8.8.8.8   # 経路確認

# 接続確認
ping -c 4 google.com
traceroute google.com  # 経路追跡
mtr google.com         # リアルタイム経路追跡

# DNS解決
dig google.com
dig google.com A @8.8.8.8  # 特定NSに問い合わせ
nslookup google.com
host google.com

# ポート・接続確認
ss -tuln               # LISTEN中のポート一覧
ss -tunp               # プロセス付き
netstat -tuln          # 旧版（非推奨）
curl -I https://example.com   # HTTPヘッダー確認
wget -O- https://example.com  # コンテンツ取得

# ファイアウォール（Ubuntu）
ufw status
ufw allow 80/tcp
ufw deny 3306
