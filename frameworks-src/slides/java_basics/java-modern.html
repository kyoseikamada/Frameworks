Stream API（Java 8+）
List names =
    List.of("Alice","Bob","Charlie","Dave");

// 中間操作（遅延評価）→ 終端操作
List result = names.stream()
    .filter(n -> n.length() > 3)    // 絞り込み
    .map(String::toUpperCase)        // 変換
    .sorted()                        // ソート
    .collect(Collectors.toList());   // 収集

// 集計
long count = names.stream()
    .filter(n -> n.startsWith("A"))
    .count();

// groupingBy
Map> byLen =
    names.stream().collect(
        Collectors.groupingBy(String::length));

// 並列（大量データ向け）
names.parallelStream()
    .map(this::heavyProcess)
    .collect(Collectors.toList());
Java 9〜21 の主要新構文
// Optional（null 安全）Java 8+
Optional.ofNullable(user)
    .map(User::getEmail)
    .orElse("unknown@example.com");

// var（型推論）Java 10+
var list = new ArrayList();

// Record（不変データクラス）Java 16+
record Point(int x, int y) {}
var p = new Point(1, 2); // p.x(), p.y()

// Sealed class Java 17+
sealed interface Shape
    permits Circle, Rectangle {}

// Pattern Matching + switch Java 21+
double area = switch (shape) {
    case Circle c    -> Math.PI * c.r() * c.r();
    case Rectangle r -> r.w() * r.h();
};
